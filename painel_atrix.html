Google Sheets — Backup automático

Cole abaixo a URL do seu Google Apps Script Web App para habilitar o envio automático dos dados de combustível para o Google Sheets.

Como configurar o Apps Script
  1. Acesse script.google.com e crie um novo projeto
  2. Cole o código abaixo e troque SEU_SHEET_ID pelo ID da planilha (na URL do Google Sheets)
  3. Salve e vá em Implantar → Nova implantação → Web App
  4. Escolha: Executar como: Eu mesmo · Acesso: Qualquer pessoa
  5. Copie a URL gerada e cole no campo acima
function doPost(e){try{var d=JSON.parse(e.postData.contents);var ss=SpreadsheetApp.openById('SEU_SHEET_ID');var ts=new Date().toLocaleString('pt-BR');var sv=ss.getSheetByName('Veiculos')||ss.insertSheet('Veiculos');sv.clearContents();sv.appendRow(['Placa','Modelo','Ano','Condutor','Setor','Superior','Tipo','Cartao','Status','Atualizado']);(d.dFuelVehicles||[]).forEach(function(v){sv.appendRow([v.placa,v.modelo,v.ano,v.condutor,v.setor,v.superior,v.tipo,v.tipoCartao,v.status,ts])});var sr=ss.getSheetByName('Solicitacoes')||ss.insertSheet('Solicitacoes');sr.clearContents();sr.appendRow(['ID','Data','Placa','Condutor','Valor','Observacao','Atualizado']);(d.dFuelRequests||[]).forEach(function(r){sr.appendRow([r.id,r.data,r.placa,r.condutor,r.valor,r.obs,ts])});return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON)}catch(err){return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.message})).setMimeType(ContentService.MimeType.JSON)}}
